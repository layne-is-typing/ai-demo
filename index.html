    <script type="module">

        import * as THREE from 'three';

        import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

        import { EffectComposer } from 'three/addons/postprocessing/EffectComposer.js';

        import { RenderPass } from 'three/addons/postprocessing/RenderPass.js';

        import { UnrealBloomPass } from 'three/addons/postprocessing/UnrealBloomPass.js';


        // --- CONFIG ---

        const CONFIG = {

            colorOnline: 0x00ffff,   // Cyber Blue

            colorOffline: 0xff0000,  // Danger Red

            colorCore: 0xff00ff,     // AI Purple

            particleCount: 400,

            flowSpeed: 0.4,

            bloomStrength: 2.5

        };


        let state = {

            isOnline: true,

            revenue: 0,

            time: 0

        };


        // --- SCENE SETUP ---

        const container = document.getElementById('canvas-container');

        const scene = new THREE.Scene();

        scene.background = new THREE.Color(0x050505);

        scene.fog = new THREE.FogExp2(0x050505, 0.02);


        const camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 1000);

        camera.position.set(0, 10, 40);


        const renderer = new THREE.WebGLRenderer({ antialias: true });

        renderer.setSize(window.innerWidth, window.innerHeight);

        renderer.setPixelRatio(window.devicePixelRatio);

        container.appendChild(renderer.domElement);


        // --- POST PROCESSING (GLOW) ---

        const renderScene = new RenderPass(scene, camera);

        const bloomPass = new UnrealBloomPass(new THREE.Vector2(window.innerWidth, window.innerHeight), 1.5, 0.4, 0.85);

        bloomPass.threshold = 0;

        bloomPass.strength = CONFIG.bloomStrength;

        bloomPass.radius = 0.5;


        const composer = new EffectComposer(renderer);

        composer.addPass(renderScene);

        composer.addPass(bloomPass);


        // --- OBJECTS ---


        // 1. The Bridge (Curve)

        class CustomSinCurve extends THREE.Curve {

            constructor(scale = 1) {

                super();

                this.scale = scale;

            }

            getPoint(t, optionalTarget = new THREE.Vector3()) {

                const tx = (t - 0.5) * 30; // -15 to 15

                const ty = Math.sin(t * Math.PI * 4) * 2; 

                const tz = 0;

                return optionalTarget.set(tx, ty, tz);

            }

        }

        const path = new CustomSinCurve(10);

        

        const tubeGeo = new THREE.TubeGeometry(path, 64, 0.5, 8, false);

        const tubeMat = new THREE.MeshBasicMaterial({ 

            color: 0x113344, 

            transparent: true, 

            opacity: 0.1, 

            wireframe: true 

        });

        const tube = new THREE.Mesh(tubeGeo, tubeMat);

        scene.add(tube);


        // 2. The Core (AI Brain) - Right Side

        const coreGeo = new THREE.IcosahedronGeometry(3, 1);

        const coreMat = new THREE.MeshBasicMaterial({ 

            color: CONFIG.colorCore, 

            wireframe: true 

        });

        const core = new THREE.Mesh(coreGeo, coreMat);

        core.position.set(15, 0, 0);

        scene.add(core);


        const coreLight = new THREE.PointLight(CONFIG.colorCore, 5, 30);

        coreLight.position.set(15, 0, 0);

        scene.add(coreLight);


        // 3. The Source (User) - Left Side

        const sourceGeo = new THREE.SphereGeometry(1, 16, 16);

        const sourceMat = new THREE.MeshBasicMaterial({ color: 0xffffff });

        const source = new THREE.Mesh(sourceGeo, sourceMat);

        source.position.set(-15, 0, 0);

        scene.add(source);



        // 4. The Disconnect Barrier

        const barrierGeo = new THREE.RingGeometry(4, 4.5, 32);

        const barrierMat = new THREE.MeshBasicMaterial({ 

            color: CONFIG.colorOffline, 

            side: THREE.DoubleSide,

            transparent: true,

            opacity: 0

        });

        const barrier = new THREE.Mesh(barrierGeo, barrierMat);

        barrier.position.set(0, 0, 0);

        barrier.rotation.y = Math.PI / 2;

        scene.add(barrier);



        // 5. Particles System

        const particleGeo = new THREE.BufferGeometry();

        const posArray = new Float32Array(CONFIG.particleCount * 3);

        const progressArray = new Float32Array(CONFIG.particleCount); 

        const speedArray = new Float32Array(CONFIG.particleCount);


        for(let i=0; i<CONFIG.particleCount; i++) {

            progressArray[i] = Math.random(); 

            speedArray[i] = 0.002 + Math.random() * 0.003;

            

            const point = path.getPoint(progressArray[i]);

            posArray[i*3] = point.x;

            posArray[i*3+1] = point.y + (Math.random()-0.5)*2; 

            posArray[i*3+2] = point.z + (Math.random()-0.5)*2; 

        }


        particleGeo.setAttribute('position', new THREE.BufferAttribute(posArray, 3));

        const particleMat = new THREE.PointsMaterial({

            color: CONFIG.colorOnline,

            size: 0.4,

            transparent: true,

            opacity: 0.8,

            blending: THREE.AdditiveBlending

        });

        const particles = new THREE.Points(particleGeo, particleMat);

        scene.add(particles);



        // --- CONTROLS ---

        const controls = new OrbitControls(camera, renderer.domElement);

        controls.enableDamping = true;

        controls.autoRotate = true;

        controls.autoRotateSpeed = 0.5;


        // --- ANIMATION LOOP ---

        function animate() {

            requestAnimationFrame(animate);


            const positions = particles.geometry.attributes.position.array;


            // Update Particles

            for(let i=0; i<CONFIG.particleCount; i++) {

                

                if (state.isOnline) {

                    progressArray[i] += speedArray[i];

                    if (progressArray[i] > 1) {

                        progressArray[i] = 0; 

                        state.revenue += 125;

                    }

                } else {

                    if (progressArray[i] < 0.45) {

                        progressArray[i] += speedArray[i];

                    } else if (progressArray[i] > 0.55) {

                        progressArray[i] += speedArray[i];

                        if (progressArray[i] > 1) progressArray[i] = 10; 

                    } else {

                        progressArray[i] = 0.5 + (Math.random()-0.5)*0.02;

                    }

                }


                if (progressArray[i] <= 1) {

                    const point = path.getPoint(progressArray[i]);

                    const time = Date.now() * 0.001;

                    const offset = 1; 

                    const angle = progressArray[i] * 20 + time;

                    

                    positions[i*3] = point.x;

                    positions[i*3+1] = point.y + Math.cos(angle) * offset;

                    positions[i*3+2] = point.z + Math.sin(angle) * offset;

                } else {

                    positions[i*3] = 9999;

                }

            }

            particles.geometry.attributes.position.needsUpdate = true;


            // Update Core

            if (state.isOnline) {

                core.rotation.x += 0.01;

                core.rotation.y += 0.02;

                const scale = 1 + Math.sin(Date.now() * 0.005) * 0.2;

                core.scale.set(scale, scale, scale);

                coreMat.wireframe = true;

                coreMat.color.setHex(CONFIG.colorCore);

                coreLight.intensity = 5;

            } else {

                core.rotation.x += 0.001;

                core.rotation.y += 0.001;

                core.scale.lerp(new THREE.Vector3(0.5, 0.5, 0.5), 0.05);

                coreMat.color.setHex(0x333333); 

                coreLight.intensity = 0.5;

            }


            // Update Barrier

            if (!state.isOnline) {

                barrier.material.opacity = Math.min(barrier.material.opacity + 0.05, 0.8);

                barrier.rotation.z += 0.05;

            } else {

                barrier.material.opacity = Math.max(barrier.material.opacity - 0.05, 0);

            }


            updateLabels();

            updateUI();


            controls.update();

            composer.render();

        }


        // --- LABELS LOGIC ---

        const labelUser = document.getElementById('label-user');

        const labelCF = document.getElementById('label-cf');

        const labelAI = document.getElementById('label-ai');

        const tempV = new THREE.Vector3();


        function updateLabels() {

            // 1. User Node

            updateLabelPosition(source.position, labelUser, 0, 2.5, 0);

            

            // 2. Cloudflare (Middle)

            updateLabelPosition(new THREE.Vector3(0,0,0), labelCF, 0, 3, 0);


            // 3. AI Node

            updateLabelPosition(core.position, labelAI, 0, 4, 0);

        }


        function updateLabelPosition(objPos, labelEl, offsetX, offsetY, offsetZ) {

            tempV.copy(objPos);

            tempV.x += offsetX;

            tempV.y += offsetY;

            tempV.z += offsetZ;


            // Project to 2D screen space

            tempV.project(camera);


            const x = (tempV.x * .5 + .5) * window.innerWidth;

            const y = (-(tempV.y * .5) + .5) * window.innerHeight;


            // Hide if behind camera

            if(tempV.z > 1) {

                labelEl.style.display = 'none';

            } else {

                labelEl.style.display = 'block';

                labelEl.style.left = `${x}px`;

                labelEl.style.top = `${y}px`;

            }

        }


        // --- UI LOGIC ---

        function updateUI() {

            const revEl = document.getElementById('revenue');

            if (state.isOnline) {

                revEl.style.color = "#00ffff";

                revEl.innerText = "$ " + state.revenue.toLocaleString();

            } else {

                revEl.style.color = "#ff3333";

                revEl.innerText = "CONNECTION LOST";

            }

        }


        window.toggleSystem = function() {

            state.isOnline = !state.isOnline;

            const btn = document.getElementById('toggle-btn');

            const statusText = document.getElementById('status-text');

            const statusDot = document.getElementById('status-dot');

            const latency = document.getElementById('latency');

            const cfLabelText = labelCF.querySelector('.label-text');


            if (state.isOnline) {

                // Go Online

                btn.innerText = "模擬拔線 (KILL SWITCH)";

                btn.className = "";

                

                statusText.innerText = "SYSTEM ONLINE";

                statusText.style.color = "#00ff00";

                statusDot.style.background = "#00ff00";

                statusDot.style.boxShadow = "0 0 8px #00ff00";

                latency.innerText = "12ms";

                latency.style.color = "#666";


                particleMat.color.setHex(CONFIG.colorOnline);

                bloomPass.strength = CONFIG.bloomStrength;


                // Label Style Reset

                labelCF.classList.remove('offline-label');

                cfLabelText.innerText = "Cloudflare (基建)";

                

            } else {

                // Go Offline

                btn.innerText = "重啟系統 (RESTART)";

                btn.className = "restart";

                

                statusText.innerText = "CRITICAL FAILURE";

                statusText.style.color = "#ff0000";

                statusDot.style.background = "#ff0000";

                statusDot.style.boxShadow = "0 0 8px #ff0000";

                latency.innerText = "TIMEOUT";

                latency.style.color = "#ff0000";


                particleMat.color.setHex(CONFIG.colorOffline);

                bloomPass.strength = 0.5;


                // Label Style Alert

                labelCF.classList.add('offline-label');

                cfLabelText.innerText = "⚠ 服務中斷 (FAILURE)";

            }

        }


        window.addEventListener('resize', () => {

            camera.aspect = window.innerWidth / window.innerHeight;

            camera.updateProjectionMatrix();

            renderer.setSize(window.innerWidth, window.innerHeight);

            composer.setSize(window.innerWidth, window.innerHeight);

        });


        animate();
